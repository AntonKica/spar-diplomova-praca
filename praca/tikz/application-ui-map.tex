      font=\ttfamily\scriptsize,
      rounded corners,
      draw,
      minimum height=1.4em,
      inner xsep=5pt,
      inner ysep=2pt,
      edge={draw, ->, >=stealth, gray!60, line width=0.4pt},
      l sep=3mm,
      s sep=1mm,
